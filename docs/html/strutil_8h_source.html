137 class Tokenizer
138 {
139 public:
140 Tokenizer(const std::string& str)
141 : m_String(str), m_Offset(0), m_Delimiters(" ") {};
142
143 Tokenizer(const std::string& str, const std::string& delimiters)
144 : m_String(str), m_Offset(0), m_Delimiters(delimiters) {};
145
146 bool nextToken() { return nextToken(m_Delimiters); };
147
148
149 bool nextToken(const std::string& delimiters)
150 {
151 // find the start charater of the next token.
152 size_t i = m_String.find_first_not_of(delimiters, m_Offset);
153 if (i == std::string::npos) {
154 m_Offset = m_String.length();
155 return false;
156 }
157
158 // find the end of the token.
159 size_t j = m_String.find_first_of(delimiters, i);
160 if (j == std::string::npos) {
161 m_Token = m_String.substr(i);
162 m_Offset = m_String.length();
163 return true;
164 }
165
166 // to intercept the token and save current position
167 m_Token = m_String.substr(i, j - i);
168 m_Offset = j;
169 return true;
170 };
171
172
173 const std::string getToken() const { return m_Token;};
174
179 void reset() { m_Offset = 0; };
180
181 protected:
182 size_t m_Offset;
183 const std::string m_String;
184 std::string m_Token;
185 std::string m_Delimiters;
186 };
